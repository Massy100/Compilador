int condicional(int x) {
    if (x > 0) {
        return 1;
    } else if (x < 0) {
        return -1;
    } else {
        return 0;
    }
}

void main() {
    int valor = 5;
    while (valor > 0) {
        valor = valor - 1;
    }
    for (int i = 0; i <= 1; i++) {
        valor = valor + 1;
    }
    int resultado = condicional(valor);
}
